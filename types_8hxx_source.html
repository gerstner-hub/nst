573 cosmos::SysString getColorName(const ColorIndex idx) const {
574
575 if (auto raw = cosmos::to_integral(idx); raw < basic_colors.size()) {
576 // returning the raw pointer here is okay since we know that
577 // they're always null terminated.
578 return basic_colors[raw].data();
579 } else if (idx >= ColorIndex::START_EXTENDED) {

581 // check for extended colors
582 if (ext < extended_colors.size())

584 }

586 // unassigned
