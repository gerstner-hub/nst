18void FontColor::load(const Theme &theme, const ColorIndex idx, cosmos::SysString name) {
19 if (name.empty()) {
20 // 256 color range
21 if (cosmos::in_range(idx, ColorIndex::START_256, ColorIndex::END_256)) {
22 load256(idx);
23 return;
24 } else {
25 name = theme.getColorName(idx);
26 }
27 }
28
29 destroy();
30
31 auto res = ::XftColorAllocName(
32 xpp::display, xpp::visual, cmap(),
33 name.raw(), this);
34
35 if (res == True) {
36 m_loaded = true;
37 return;
38 }
39
40 auto colorname = theme.getColorName(idx);
41
42 cosmos_throw (cosmos::RuntimeError(
43 cosmos::sprintf("could not allocate color %d ('%s')", cosmos::to_integral(idx),
44 colorname.empty() ? "unknown" : colorname.raw())));
45}

47void FontColor::load256(const ColorIndex idx) {
48 /*
49 * xterm 256 color support has the following planes:
50 *
51 * 0 - 15: the 16 standard system colors which aren't handled here
52 * 16 - 232: extended color palette organized in three sub-planes
53 * 232 - 255: extended greyscale colors
54 */
55
57 auto sixd_to_16bit = [](size_t x) -> uint16_t {
58 return x == 0 ? 0 : 0x3737 + 0x2828 * x;
59 };
60
61 XRenderColor tmp = { 0, 0, 0, 0xffff };
62
63 if (idx < ColorIndex::START_GREYSCALE) { // same colors as xterm
64 auto baseindex = cosmos::to_integral(idx - ColorIndex::START_256);
65 tmp.red = sixd_to_16bit((baseindex/36) % 6);
66 tmp.green = sixd_to_16bit((baseindex/ 6) % 6);
67 tmp.blue = sixd_to_16bit((baseindex/ 1) % 6);
68 } else { // greyscale
69 auto baseindex = cosmos::to_integral(idx - ColorIndex::START_GREYSCALE);
70 tmp.red = 0x0808 + 0x0a0a * baseindex;
71 tmp.green = tmp.blue = tmp.red;
72 }
73
74 load(tmp);
75}

111void RenderColor::setFromRGB(const ColorIndex rgb) {
112 /* The X color values are 16-bit wide and we need to
113 * translate the one color bytes into the upper byte in the
114 * XRenderColor */
115 auto raw = cosmos::to_integral(rgb);
116 alpha = 0xffff;
117 red = (raw & 0xff0000) >> 8;
118 green = (raw & 0xff00);
119 blue = (raw & 0xff) << 8;
120}

165void ColorManager::configureFor(const Glyph base) {
166 auto assignBaseColor = [this](FontColor &out, const ColorIndex color) {
167 if (is_true_color(color)) {
168 out.load(RenderColor{color});
169 } else {
170 // color is a palette index
171 out = fontColor(color);
172 }
173 };
174
175 assignBaseColor(m_front_color, base.fg);
176 assignBaseColor(m_back_color, base.bg);
177
178 // Change basic system colors [0-7] to bright system colors [8-15]
179 if (base.needBrightColor() && base.isBasicColor()) {
180 m_front_color = fontColor(base.toBrightColor());
181 } else if (base.needFaintColor()) {
182 m_front_color.makeFaint();
183 }
184
185 if (m_twin.inReverseMode()) {
186 applyReverseMode();
187 }
188
189 if (base.useReverseColor()) {
190 std::swap(m_front_color, m_back_color);
191 }
192
193 if (base.mode[Attr::BLINK] && m_twin.inBlinkMode()) {
194 m_front_color = m_back_color;
195 } else if (base.mode[Attr::INVISIBLE]) {
196 m_front_color = m_back_color;
197 }
198}

200void ColorManager::applyReverseMode() {
201 // if one of the colors is the default color then switch to the
202 // reverse default color, otherwise simply invert the raw color value
203
204 if (m_front_color == defaultFront()) {
205 m_front_color = defaultBack();
206 } else {
207 m_front_color.invert();
208 }
209
210 if (m_back_color == defaultBack()) {
211 m_back_color = defaultFront();
212 } else {
213 m_back_color.invert();
214 }

217const FontColor& ColorManager::applyCursorColor(const bool is_selected, Glyph &glyph) const {
218 // Select the right color for the right mode.
219 glyph.mode.limit({Attr::BOLD, Attr::ITALIC, Attr::UNDERLINE, Attr::STRUCK, Attr::WIDE});
220
221 if (m_twin.inReverseMode()) {
222 glyph.setReverseColor();
223 glyph.bg = m_theme.fg;
224 if (is_selected) {
225 glyph.fg = m_theme.reverse_cursor_color;
226 return fontColor(m_theme.cursor_color);
227 } else {
228 glyph.fg = m_theme.cursor_color;
229 return fontColor(m_theme.reverse_cursor_color);
230 }
231 } else {
232 if (is_selected) {
233 glyph.fg = m_theme.fg;
234 glyph.bg = m_theme.reverse_cursor_color;
235 } else {
236 if (m_twin.getCursorStyle() == CursorStyle::REVERSE_BLOCK) {
237 glyph.setReverseColor();
238 return fontColor(m_theme.cursor_color);
239 } else {
240 glyph.fg = m_theme.bg;
241 glyph.bg = m_theme.cursor_color;
242 }
243 }
244
245 return fontColor(glyph.bg);
246 }
247}
