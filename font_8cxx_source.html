149bool Font::load(const FontPattern &pattern) {
150 unload();
151 // Manually configure instead of calling XftMatchFont so that we can
152 // use the configured pattern for "missing glyph" lookups.
153 auto &display = xpp::display;
154 FcPattern *configured = FcPatternDuplicate(pattern.raw());
155 if (!configured) {
156 return false;
157 } else {
158 FcPatternGuard configured_guard{configured};
159 FcConfigSubstitute(nullptr, configured, FcMatchPattern);
160 ::XftDefaultSubstitute(display, xpp::raw_screen(display.defaultScreen()), configured);
161
162 FcResult result;
163 FcPattern *match = FcFontMatch(nullptr, configured, &result);
164 if (!match)
165 return false;
166
167 FcPatternGuard match_guard{match};
168
169 if (!(m_match = ::XftFontOpenPattern(display, match)))
170 return false;
171
172 m_pattern = configured;
173
174 // ownership has been transferred to m_pattern
175 configured_guard.disarm();
176 // ... and to m_match
177 match_guard.disarm();
178 }
179
180 checkSlant(pattern);
181 checkWeight(pattern);
182
183 XGlyphInfo extents;
184 ::XftTextExtentsUtf8(display, m_match,
185 (const FcChar8*)config::ASCII_PRINTABLE.data(),
186 config::ASCII_PRINTABLE.size(), &extents);
187
188 // pixels above and below the baseline of a character make up the
189 // character height
190 m_height = ascent() + descent();
191 m_width = (extents.xOff + config::ASCII_PRINTABLE.size() - 1) / config::ASCII_PRINTABLE.size();
192
193 return true;
194}

196void Font::checkSlant(const FontPattern &pattern) {
197 const FontPattern opened = this->pattern();
198
199 if (auto wanted_slant = pattern.getSlant(); wanted_slant != std::nullopt) {
200 // Check if xft was unable to find a font with the appropriate
201 // slant but gave us one anyway. Try to mitigate.
202 if (auto our_slant = opened.getSlant(); our_slant != std::nullopt && *our_slant < *wanted_slant) {
203 m_bad_slant = true;
204 std::cerr << "nst: font slant does not match\n";
205 }
206 }
207}

220FcPattern* Font::queryFontConfig(const Rune rune) const {
221 FcResult fc_res;
222 if (!m_set) {
223 m_set = FcFontSort(nullptr, m_pattern, /*trim=*/FcTrue, nullptr, &fc_res);
224
225 if (!m_set)
226 return nullptr;
227 }
228 FcFontSet *fc_sets[] = { m_set };
229
230 // Nothing was found in the cache. Now use some dozen of Fontconfig
231 // calls to get the font for one single character.
232 //
233 // Xft and fontconfig are design failures.
234 FcPattern *fc_pattern = FcPatternDuplicate(m_pattern);
235 FcPatternGuard fc_pattern_guard{fc_pattern};
236 FcCharSet *fc_charset = FcCharSetCreate();
237 FcCharSetGuard fc_charset_guard{fc_charset};
238
239 FcCharSetAddChar(fc_charset, rune);
240 FcPatternAddCharSet(fc_pattern, FC_CHARSET, fc_charset);
241 FcPatternAddBool(fc_pattern, FC_SCALABLE, FcTrue);
242
243 FcConfigSubstitute(nullptr, fc_pattern, FcMatchPattern);
244 FcDefaultSubstitute(fc_pattern);
245
246 return FcFontSetMatch(nullptr, fc_sets, cosmos::num_elements(fc_sets), fc_pattern, &fc_res);
247}

305bool FontManager::loadFonts() {
306 FontPattern pattern{m_font_spec};
307
308 if (!pattern.valid())
309 return false;
310
311 unloadFonts();
312
313 if (m_used_font_size) {
314 pattern.setPixelSize(*m_used_font_size);
315 } else {
316 if (auto pxsize = pattern.pixelSize(); pxsize.has_value())
317 m_used_font_size = pxsize;
318 else if(auto ptsize = pattern.pointSize(); ptsize.has_value())
319 // leave to be determined after loading the first font below
320 ;
321 else {
322 // Use default font size, if none given. This is to
323 // have a known m_used_font_size value.
324 m_used_font_size = config::FONT_DEFAULT_SIZE_PX;
325 pattern.setPixelSize(*m_used_font_size);
326 }
327 m_default_font_size = m_used_font_size;
328 }
329
330 if (!m_normal_font.load(pattern))
331 return false;
332
333 if (!m_used_font_size) {
334 auto loaded = m_normal_font.pattern();
335 if (auto pxsize = loaded.pixelSize(); pxsize.has_value()) {
336 m_used_font_size = *pxsize;
337 if (!m_default_font_size)
338 m_default_font_size = *pxsize;
339 }
340 }
341
342 pattern.setSlant(Slant::ITALIC);
343 if (!m_italic_font.load(pattern))
344 return false;
345
346 pattern.setWeight(Weight::BOLD);
347 if (!m_italic_bold_font.load(pattern))
348 return false;
349
350 pattern.setSlant(Slant::ROMAN);
351 if (!m_bold_font.load(pattern))
352 return false;
353
354 return true;
355}

375// font cannot be const since libXft doesn't mark it const in XftCharIndex,
376// also the implementation actually changes data in it (caches)
377std::tuple<XftFont*, FT_UInt> FontManager::lookupFontEntry(const Rune rune, Font &font) {
378 // Lookup character index with default font.
379 auto glyphidx = ::XftCharIndex(xpp::display, font.match(), rune);
380 if (glyphidx) {
381 return std::make_tuple(font.match(), glyphidx);
382 }
383
384 // Fallback on font cache, search the font cache for match.
385 for (auto &entry: m_font_cache) {
386 glyphidx = ::XftCharIndex(xpp::display, entry.font, rune);
387 if (glyphidx && entry.flags == font.flags()) {
388 // Everything correct.
389 return std::make_tuple(entry.font, glyphidx);
390 } else if (!glyphidx && entry.flags == font.flags() && entry.rune == rune) {
391 // We got a default font for a not found glyph.
392 return std::make_tuple(entry.font, glyphidx);
393 }
394 }
395
396 // Nothing was found. Use fontconfig to find matching font.
397 auto pattern = font.queryFontConfig(rune);
398
399 // Allocate memory for the new cache entry.
400 auto new_font = ::XftFontOpenPattern(xpp::display, pattern);
401 if (!new_font) {
402 cosmos_throw (cosmos::ApiError("XftFontOpenPattern() failed seeking fallback font"));
403 }
404
405 m_font_cache.emplace_back(FontCache{new_font, font.flags(), rune});
406
407 glyphidx = ::XftCharIndex(xpp::display, new_font, rune);
408
409 return std::make_tuple(new_font, glyphidx);
410}

412void FontManager::sanitize(Glyph &g) const {
413 // Fallback on color display for attributes not supported by the font
414 if (g.mode[Attr::ITALIC] && g.mode[Attr::BOLD]) {
415 if (m_italic_bold_font.hasBadSlant() || m_italic_bold_font.hasBadWeight()) {
416 g.fg = config::DEFAULT_ATTR;
417 }
418 } else if ((g.mode[Attr::ITALIC] && m_italic_font.hasBadSlant()) ||
419 (g.mode[Attr::BOLD] && m_bold_font.hasBadWeight())) {
420 g.fg = config::DEFAULT_ATTR;
421 }
