23std::vector<MouseShortcut> get_mouse_shortcuts(Nst &nst) {
24
25 auto &tty = nst.tty();
26 auto &wsys = nst.wsys();
27 auto &term = nst.term();
28
29 auto ttysend = [&](const std::string_view s) {
30 tty.write(s, TTY::MayEcho{true});
31 };
32
33 using xpp::Button;
34
35 auto scrollHistoryUp = [&term, ttysend](bool shift) {
36 // on the alt screen we inject special escape sequences that
37 // are used by programs like less and vim for scrolling.
38 if (term.onAltScreen()) {
39 ttysend("\031");
40 } else {
41 if (shift)
42 term.scrollHistoryUpByPage(0.5);
43 else
44 term.scrollHistoryUpByLines(+5);
45 }
46 };
47
48 auto scrollHistoryDown = [&term, ttysend](bool shift) {
49 // see above
50 if (term.onAltScreen()) {
51 ttysend("\005");
52 } else {
53 if (shift)
54 term.scrollHistoryDownByPage(0.5);
55 else
56 term.scrollHistoryDownByLines(+5);
57 }
58 };
59
60 return {
61 // mask button function release
62 MouseShortcut{ Mask{Mod::ANY}, Button::BUTTON2, std::bind(&WindowSystem::pasteSelection, &wsys), true, StopScrolling{true}},
63 // change font size similar to how it works in web browsers
64 MouseShortcut{ Mask{Mod::CONTROL}, Button::BUTTON4, std::bind(&WindowSystem::zoomFont, &wsys, +1), false },
65 MouseShortcut{ Mask{Mod::CONTROL}, Button::BUTTON5, std::bind(&WindowSystem::zoomFont, &wsys, -1), false },
66 // these generate sequences used by less/vi for scrolling
67 MouseShortcut{ Mask{Mod::SHIFT}, Button::BUTTON4, std::bind(scrollHistoryUp, true), false },
68 MouseShortcut{ Mask{Mod::SHIFT}, Button::BUTTON5, std::bind(scrollHistoryDown, true), false },
69 // regular scrolling via mouse wheel
70 MouseShortcut{ Mask{Mod::ANY}, Button::BUTTON4, std::bind(scrollHistoryUp, false), false },
71 MouseShortcut{ Mask{Mod::ANY}, Button::BUTTON5, std::bind(scrollHistoryDown, false), false },
72 };
73}

76std::vector<KbdShortcut> get_kbd_shortcuts(Nst &nst) {
77 //constexpr auto MODKEY = Mod1Mask;
78 constexpr xpp::InputMask TERMMOD{Mod::CONTROL, Mod::SHIFT};
79
80 auto &tty = nst.tty();
81 auto &wsys = nst.wsys();
82 auto &term = nst.term();
83 auto selPaste = std::bind(&WindowSystem::pasteSelection, &wsys);
84
85 auto togglePrinter = [&]() { term.setPrintMode(!term.isPrintMode()); };
86 auto printScreen = [&]() { term.dump(); };
87 auto printSel = [&]() { nst.selection().dump(); };
88
89 return {
90 // mask keysym function
91 { Mask{Mod::ANY}, KeyID::BREAK, std::bind(&TTY::sendBreak, &tty) },
92 { Mask{Mod::CONTROL}, KeyID::PRINT, togglePrinter },
93 { Mask{Mod::SHIFT}, KeyID::PRINT, printScreen },
94 { Mask{Mod::ANY}, KeyID::PRINT, printSel },
95 { Mask{Mod::SHIFT,
96 Mod::MOD1}, KeyID::PRIOR, std::bind(&WindowSystem::zoomFont, &wsys, +1), "font_zoom_in" },
97 { Mask{Mod::SHIFT,
98 Mod::MOD1}, KeyID::NEXT, std::bind(&WindowSystem::zoomFont, &wsys, -1), "font_zoom_out" },
99 { Mask{Mod::SHIFT,
100 Mod::MOD1}, KeyID::HOME, std::bind(&WindowSystem::resetFont, &wsys), "font_zoom_reset" },
101 { TERMMOD, KeyID::C, std::bind(&WindowSystem::copyToClipboard, &wsys), "copy_to_clipboard" },
102 { TERMMOD, KeyID::V, std::bind(&WindowSystem::pasteClipboard, &wsys), "paste_clipboard", StopScrolling{true} },
103 { TERMMOD, KeyID::Y, selPaste, "paste_selection", StopScrolling{true} },
104 { Mask{Mod::SHIFT}, KeyID::INSERT, selPaste, "", StopScrolling{true} },
105 { TERMMOD, KeyID::NUM_LOCK, std::bind(&WindowSystem::toggleNumlock, &wsys) },
106 { Mask{Mod::SHIFT}, KeyID::PRIOR, std::bind(&Term::scrollHistoryUpByLines, &term, +10), "scroll_history_up_slow" },
107 { Mask{Mod::SHIFT}, KeyID::NEXT, std::bind(&Term::scrollHistoryDownByLines, &term, +10), "scroll_history_down_slow" },
108 { TERMMOD, KeyID::PRIOR, std::bind(&Term::scrollHistoryUpByPage, &term, +0.5), "scroll_history_up_fast" },
109 { TERMMOD, KeyID::NEXT, std::bind(&Term::scrollHistoryDownByPage, &term, +0.5), "scroll_history_down_fast" },
110 { TERMMOD, KeyID::HOME, std::bind(&Term::scrollHistoryUpMax, &term), "scroll_history_up_max" },
111 { TERMMOD, KeyID::END, std::bind(&Term::stopScrolling, &term), "stop_scrolling" },
112 { TERMMOD, KeyID::B, std::bind(&Nst::pipeBufferToExternalCommand, &nst), "open_buffer_in_editor" },
113 { Mask{}, KeyID::F11, std::bind(&WindowSystem::toggleFullScreen, &wsys), "toggle_fullscreen" },
114 { TERMMOD, KeyID::KP_DIVIDE, std::bind(&WindowSystem::invertColors, &wsys), "invert_colors" }
115 };
