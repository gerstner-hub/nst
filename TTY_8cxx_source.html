35cosmos::FileDescriptor TTY::create(const Extent extent) {
36 const auto &cmdline = m_nst.cmdline();
37
38 if (m_cmd_file.isOpen()) {
39 cosmos_throw (cosmos::UsageError("TTY has already been created"));
40 }
41
42 setupIOFile(cmdline.iofile.getValue());
43
44 if (auto &line = cmdline.tty_line.getValue(); !line.empty()) {
45 // operate on a real TTY line, running stty on it
46 openTTY(line);
47 } else {
48 createPTY(extent);
49 }
50
51 m_terminal.setFD(m_cmd_file);
52 setupPoller();
53
54 return m_cmd_file.fd();
55}

97void TTY::setupIOFile(const std::string &path) {
98 m_io_file.close();
99
100 if (path == "-") {
101 m_io_file.open(cosmos::stdout, cosmos::AutoCloseFD{false});
102 } else if (!path.empty()) {
103 try {
104 m_io_file.open(
105 path,
106 cosmos::OpenMode::WRITE_ONLY,
107 {cosmos::OpenFlag::CREATE, cosmos::OpenFlag::TRUNCATE},
108 cosmos::FileMode(cosmos::ModeT{0640})
109 );
110 } catch (const std::exception &ex) {
111 m_nst.logger().error() << "Error opening " << path << ": " << ex.what() << "\n";
112 }
113 }
114
115 m_nst.term().setPrintMode(m_io_file.isOpen());

140size_t TTY::read() {
141 try {
142 size_t read_bytes;
143
144 try {
145 read_bytes = m_cmd_file.read(m_buf + m_buf_bytes, sizeof(m_buf) - m_buf_bytes);
146 } catch (const cosmos::ApiError &ex) {
147 if (ex.errnum() == cosmos::Errno::IO_ERROR) {
148 // the way the PTY is operated currently
149 // causes no EOF condition to be signaled but
150 // an EIO is returned. There are different
151 // modes the PTY can be operated in, but for
152 // the moment let's catch the EIO and
153 // translate it into EOF.
154 return 0;
155 }
156
157 throw;
158 }
159
160 if (read_bytes == 0) {
161 // EOF, never happens, see above
162 return 0;
163 }
164
165 // append read bytes to unprocessed bytes
166 m_buf_bytes += read_bytes;
167 const auto view = std::string_view{m_buf, m_buf_bytes};
168 const auto written = m_nst.term().write(view, Term::ShowCtrlChars(false));
169 m_buf_bytes -= written;
170 // keep any incomplete UTF-8 byte sequence for the next call
171 if (m_buf_bytes > 0)
172 // NOTE: using a ring buffer (e.g. via memmap) we could
173 // probably avoid this memmove with the trade off of
174 // added complexity.
175 std::memmove(m_buf, m_buf + written, m_buf_bytes);
176 return read_bytes;
177 } catch (const std::exception &ex) {
178 cosmos_throw (cosmos::RuntimeError(cosmos::sprintf("Couldn't read from shell: %s", ex.what())));
179 return 0;
180 }
181}

183void TTY::write(const std::string_view sv, const MayEcho echo) {
184
185 auto &term = m_nst.term();
186 const auto mode = term.mode();
187
188 if (echo && mode[Term::Mode::TECHO])
189 // display data on screen
190 term.write(sv, Term::ShowCtrlChars{true});
191
192 if (!mode[Term::Mode::CRLF]) {
193 // forward unmodified data to child
194 writeRaw(sv);
195 return;
196 }
197
198 // otherwise we need to translate newlines
199
200 // This is similar to how the kernel handles ONLCR for ttys
201 for (auto it = sv.begin(); it < sv.end();) {
202 if (*it == '\r') {
203 it++;
204 writeRaw("\r\n");
205 } else {
206 // write the segment up to the next CR
207 auto next = std::find(it, sv.end(), '\r');
208 writeRaw({&(*it), static_cast<size_t>(next - it)});
209 it = next;
210 }
211 }
212}

214void TTY::writeRaw(const std::string_view sv) {
215 // Remember that we are potentially using a real TTY, which might be a modem line.
216 // Writing too much will clog the line. That's why we are doing this dance.
217 using Event = cosmos::Poller::Event;
218 const char *data = sv.data();
219 size_t written;
220
221 for (size_t limit = 256, left = sv.size(); left > 0; ) {
222 for (const auto &event: m_cmd_poller.wait()) {
223
224 const auto events = event.getEvents();
225
226 if (events & Event::OUTPUT_READY) {
227 // Only write the bytes written by write() or the
228 // default of 256. This seems to be a reasonable value
229 // for a serial line. Bigger values might clog the I/O.
230 //
231 // TODO: since Modem lines are not likely to
232 // be used any more this is causing a lot of
233 // system call overhead in case of a PTY.
234 // Using a simpler algorithm for PTYs might be
235 // more efficient.
236 written = m_cmd_file.write(data, std::min(left, limit));
237 if (written == left)
238 // All bytes have been written.
239 return;
240
241 // We weren't able to write out everything.
242 // This means the buffer is getting full
243 // again. Empty it.
244 if (left < limit)
245 limit = this->read();
246 left -= written;
247 data += written;
248 }
249
250 // NOTE: the order of output/input is important, we
251 // need to prefer writes, otherwise we clog our own
252 // input buffer until it's full, and nothing is ever
253 // written out.
254 if (events & Event::INPUT_READY) {
255 limit = this->read();
256 }
257 }
258 }

264 // according to the man page these fields are unused on Linux, but it

285void TTY::executeShell(cosmos::FileDescriptor slave) {
286
287 cosmos::PasswdInfo pw_info{cosmos::proc::get_real_user_id()};
288 if (!pw_info.valid()) {
289 cosmos_throw (cosmos::InternalError("who are you?"));
290 }
291
292 cosmos::SysString shell;
293 if (auto sh = std::getenv("SHELL"); sh != nullptr)
294 shell = sh;
295 else
296 // try the shell from passwd
297 shell = pw_info.shell();
298
299 // if still empty then use compile time default
300 if (shell.empty()) {
301 shell = config::SHELL.data();
302 }
303
304 cosmos::ChildCloner cloner;
305
306 // code executed in the child before we execute the new program
307 cloner.setPostForkCB([this, &slave, pw_info, shell](const cosmos::ChildCloner &) {
308 // close unnecessary file descriptors in the child
309 m_io_file.close();
310 m_cmd_file.close();
311
312 // create a new process group
313 cosmos::proc::create_new_session();
314
315 // make the slave end of the TTY the new default file descriptors for the child
316 for (auto stdfd: {cosmos::stdin, cosmos::stdout, cosmos::stderr}) {
317 slave.duplicate(stdfd, cosmos::CloseOnExec(false));
318 }
319
320 // make our new TTY the controlling terminal of the child
321 cosmos::Terminal(slave).makeControllingTerminal();
322
323 // make sure no unnecessary duplicate of the slave TTY exists
324 if (slave.raw() > cosmos::FileNum{2})
325 slave.close();
326
327 using namespace cosmos;
328
329 // restore default signal handlers
330 for (auto sig: {
331 signal::CHILD, signal::HANGUP, signal::INTERRUPT,
332 signal::QUIT, signal::TERMINATE, signal::ALARM}) {
333 cosmos::signal::restore(cosmos::Signal{sig});
334 }
335
336 for (const auto var: {"COLUMNS", "LINES", "TERMCAP"}) {
337 unsetenv(var);
338 }
339
340 const proc::OverwriteEnv overwrite{true};
341
342 proc::set_env_var("LOGNAME", pw_info.name(), overwrite);
343 proc::set_env_var("USER", pw_info.name(), overwrite);
344 proc::set_env_var("SHELL", shell, overwrite);
345 proc::set_env_var("HOME", pw_info.homeDir(), overwrite);
346 proc::set_env_var("TERM", config::TERM_NAME.data(), overwrite);
347 });
348
349 if (auto &args = m_nst.cmdline().rest.getValue(); !args.empty()) {
350 cloner.setArgs(args);
351 } else {
352 // use default configuration
353 if (!config::SCROLL.empty()) {
354 cloner.setArgsFromView({config::SCROLL, config::UTMP.empty() ? shell : config::UTMP});
355 } else if (!config::UTMP.empty()) {
356 cloner.setExe(config::UTMP);
357 } else {
358 cloner.setExe(shell);
359 }
360 }
361
362 // this may throw, we'll let it pass through to the caller
363 m_child_proc = cloner.run();
364}

366void TTY::handleSigChildEvent() {
367 auto res = m_child_proc.wait();
368
369 if (!res.exitedSuccessfully()) {
370 cosmos_throw (cosmos::RuntimeError(
371 cosmos::sprintf("child exited with status %d",
372 cosmos::to_integral(res.exitStatus()))));
373 } else if (res.signaled()) {
374 cosmos_throw (cosmos::RuntimeError(
375 cosmos::sprintf("child terminated due to signal %d",
376 cosmos::to_integral(res.termSignal().raw()))));
377 }
378}
