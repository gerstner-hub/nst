46void StringEscape::oscColorResponse(const ColorIndex idx, const int code) {
47 uint8_t r, g, b;
48
49 if (!m_nst.wsys().colorManager().toRGB(idx, r, g, b)) {
50 std::cerr << "erresc: failed to fetch osc color " << cosmos::to_integral(idx) << "\n";
51 return;
52 }
53
54 std::string res;
55
56 if (code == 4) {
57 // for code 4 type responses also the index is reported back
58 res = cosmos::sprintf("\033]4;%d;rgb:%02x%02x/%02x%02x/%02x%02x\007", cosmos::to_integral(idx), r, r, g, g, b, b);
59 } else {
60 res = cosmos::sprintf("\033]%d;rgb:%02x%02x/%02x%02x/%02x%02x\007", code, r, r, g, g, b, b);
61 }
62
63 if (res.empty()) {
64 // sprintf error occured
65 std::cerr << "error: sprintf failed while printing OSC (code " << code << ") response\n";
66 return;
67 }
68
69 m_nst.tty().write(res, TTY::MayEcho{true});
70}

88void StringEscape::process() {
89 parseArgs();
90
91 switch (m_esc_type) {
92 case Type::OSC: // Operating System Command
93 if (!processOSC()) {
94 // error in OSC command, dump
95 dump("erresc: unknown str escape");
96 }
97 return;
98 case Type::SET_TITLE: // old title set compatibility
99 setTitle(m_args.empty() ? "" : m_args[0].data());
100 return;
101 case Type::DCS: //
102 case Type::APC: // Application Program Command
103 case Type::PM: // Privacy Message
104 case Type::NONE: // should never happen
105 return;
106 }
107}

109bool StringEscape::processOSC() {
110 auto &wsys = m_nst.wsys();
111 const int par = m_args.empty() ? 0 : std::atoi(m_args[0].data());
112 const auto numargs = m_args.size();
113
114 // handles different color settings and reporting
115 auto handle_color = [&](const std::string_view label, const int code, const ColorIndex idx) {
116 if (numargs < 2)
117 return false;
118
119 const auto arg = m_args[1];
120
121 if (arg == "?")
122 // report current color setting
123 oscColorResponse(idx, code);
124 else if (!wsys.colorManager().setColorName(idx, arg.data()))
125 std::cerr << "erresc: invalid " << label << " color: " << arg << "\n";
126 else
127 m_nst.term().setAllDirty();
128
129 return true;
130 };
131
132 constexpr int SET_COLOR_INDEX = 4;
133 constexpr int RESET_COLOR_INDEX = 104;
134
135 // for reference see: https://www.xfree86.org/current/ctlseqs.html
136 switch (par) {
137 case 0: // change icon name _and_ window title
138 if (numargs > 1) {
139 const auto title = m_args[1].data();
140 setTitle(title);
141 setIconTitle(title);
142 }
143 break;
144 case 1: // change icon name
145 if (numargs > 1)
146 setIconTitle(m_args[1].data());
147 break;
148 case 2: // change window title
149 if (numargs > 1)
150 setTitle(m_args[1].data());
151 break;
152 case 52: // manipulate selection data
153 if (numargs > 2 && config::ALLOW_WINDOW_OPS) {
154 auto decoded = base64::decode(m_args[2]);
155 if (!decoded.empty()) {
156 wsys.selection().setSelection(decoded);
157 wsys.copyToClipboard();
158 } else {
159 std::cerr << "erresc: invalid base64\n";
160 }
161 }
162 break;
163 case 10: // change text FG color
164 return handle_color("foreground", par, config::DEFAULT_FG);
165 case 11: // change text BG color
166 return handle_color("background", par, config::DEFAULT_BG);
167 case 12: // change text cursor color
168 return handle_color("cursor", par, config::DEFAULT_CURSOR_COLOR);
169 case SET_COLOR_INDEX: // change color number to RGB value
170 if (numargs < 3)
171 return false;
172 /* FALLTHROUGH */
173 case RESET_COLOR_INDEX: { // color reset
174 const auto name = (par == SET_COLOR_INDEX ? m_args[2] : std::string_view(""));
175 const int rawindex = (numargs > 1) ? atoi(m_args[1].data()) : -1;
176 const auto colindex = rawindex >= 0 ? ColorIndex(rawindex) : ColorIndex::INVALID;
177
178 if (name == "?")
179 oscColorResponse(colindex, SET_COLOR_INDEX);
180 else if (!wsys.colorManager().setColorName(colindex, name.data())) {
181 if (par == RESET_COLOR_INDEX && numargs <= 1) {
182 wsys.resetColors();
183 break; // color reset without parameter
184 }
185 std::cerr << "erresc: invalid color index=" << rawindex << ", name=" << (name.empty() ? "(null)" : name) << "\n";
186 } else {
187 // TODO if defaultbg color is changed, borders are dirty
188 m_nst.term().setAllDirty();
189 }
190 break;
191 }
192 default: return false;
193 }
194
195 return true;
196}

199void StringEscape::parseArgs() {
200 auto it = m_str.begin();
201
202 // parameters are separated by semilocon, extract them
203
204 while (it != m_str.end() && m_args.size() < MAX_STR_ARGS) {
205 auto end = std::find(it, m_str.end(), ';');
206
207 // NOTE: C++20 has a better constructor using iterators
208 std::string_view sv(&(*it), std::distance(it, end));
209 m_args.push_back(sv);
210
211 if (end != m_str.end()) {
212 // make sure the views we add to m_args are properly terminated
213 *end = '\0';
214 end++;
215 }
216
217 it = end;
218 }
219
220 if (it != m_str.end()) {
221 std::cerr << __FUNCTION__ << ": maximum number of arguments exceeded\n";
222 }
223}

225void StringEscape::dump(const std::string_view prefix) const {
226 std::cerr << prefix << " ESC" << static_cast<char>(m_esc_type);
227
228 for (const auto c: m_str) {
229 if (c == '\0') {
230 std::cerr << '\n';
231 return;
232 } else if (cosmos::printable(c)) {
233 std::cerr << (char)c;
234 } else if (c == '\n') {
235 std::cerr << "(\\n)";
236 } else if (c == '\r') {
237 std::cerr << "(\\r)";
238 } else if (c == 0x1b) {
239 std::cerr << "(\\e)";
240 } else {
241 std::cerr << "(" << cosmos::HexNum{static_cast<unsigned>(c), 2}.showBase(false) << ")";
242 }
243 }
244 std::cerr << "ESC\\\n";
245}

254void StringEscape::add(const std::string_view s) {
255 if (m_str.size() + s.size() >= m_str.capacity()) {
256 /*
257 * Here is a bug in terminals. If the user never sends
258 * some code to stop the str or esc command, then nst
259 * will stop responding. But this is better than
260 * silently failing with unknown characters. At least
261 * then users will report back.
262 *
263 * In the case users ever get fixed, here is the code:
264 */
265 /*
266 * m_nst.term().m_esc_state.reset();
267 * process();
268 */
269 if (m_str.size() > (SIZE_MAX - utf8::UTF_SIZE) / 2)
270 return;
271 m_str.reserve(m_str.capacity() << 1);
272 }
273
274 m_str.append(s);
275}
