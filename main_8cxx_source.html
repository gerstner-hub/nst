80cosmos::ExitStatus IpcClient::main(const int argc, const char **argv) {
81 m_cmdline.parse(argc, argv);
82
83 const auto request = [this]() -> Message {
84 if (m_cmdline.save_snapshot.isSet())
85 return Message::SNAPSHOT_HISTORY;
86 else if (m_cmdline.get_snapshot.isSet())
87 return Message::GET_SNAPSHOT;
88 else if (m_cmdline.get_history.isSet())
89 return Message::GET_HISTORY;
90 else if (m_cmdline.test_connection.isSet())
91 return Message::PING;
92 else {
93 throw INT_ERR;
94 }
95 }();
96
97 auto connection = connect();
98 connection.send(&request, sizeof(request));
99 receiveData(request, connection);
100 return cosmos::ExitStatus::SUCCESS;
101}

103cosmos::UnixConnection IpcClient::connect() {
104 constexpr auto envvar = "NST_IPC_ADDR";
105 const auto ipc_addr = cosmos::proc::get_env_var(envvar);
106
107 if (!ipc_addr) {
108 if (!m_cmdline.test_connection.isSet()) {
109 std::cerr << "Environment variable '" << envvar << "' is not set. Cannot connect to nst.\n";
110 }
111
112 throw CONN_ERR;
113 }
114
115 try {
116 return m_sock.connect(cosmos::UnixAddress{*ipc_addr, cosmos::UnixAddress::Abstract{true}});
117 } catch (const cosmos::ApiError &error) {
118 if (!m_cmdline.test_connection.isSet()) {
119 std::cerr << "Failed to connect to nst (address: @" << *ipc_addr << "): " << error.what() << "\n";
120 }
121
122 throw CONN_ERR;
123 }
124}

126void IpcClient::receiveData(const Message request, cosmos::UnixConnection &connection) {
127 std::string buffer;
128 while (true) {
129 buffer.resize(IpcHandler::MAX_CHUNK_SIZE);
130 auto len = connection.receive(buffer.data(), buffer.size(), cosmos::MessageFlags{cosmos::MessageFlag::TRUNCATE});
131 if (len == 0) {
132 if (m_cmdline.test_connection.isSet()) {
133 // should have received a PING reply
134 throw INT_ERR;
135 }
136 return;
137 } else if(len > buffer.size()) {
138 std::cerr << "IPC packet was truncated from " << len << " to " << buffer.size() << "!\n";
139 len = buffer.size();
140 }
141
142 buffer.resize(len);
143
144 if (request == Message::SNAPSHOT_HISTORY) {
145 std::cerr << "received unexpected data as reply to SNAPSHOT_HISTORY\n";
146 throw INT_ERR;
147 } else if (request == Message::PING) {
148 Message reply;
149 if (buffer.size() != sizeof(reply)) {
150 std::cerr << "received bad reply len for PING message\n";
151 throw INT_ERR;
152 }
153
154 std::memcpy(&reply, buffer.data(), sizeof(reply));
155 if (reply != Message::PING) {
156 std::cerr << "received bad PING reply message\n";
157 throw INT_ERR;
158 }
159
160 // PING test succeeded
161 return;
162 } else {
163 // simply forward data to stdout
164 std::cout << buffer;
165 }
166 }
167}
