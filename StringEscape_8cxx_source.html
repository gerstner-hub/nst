46void StringEscape::oscColorResponse(const ColorIndex idx, const int code) {
47 uint8_t r, g, b;
48
49 if (!m_nst.wsys().colorManager().toRGB(idx, r, g, b)) {
50 m_nst.logger().error() << "erresc: failed to fetch osc color " << cosmos::to_integral(idx) << "\n";
51 return;
52 }
53
54 std::string res;
55
56 if (code == 4) {
57 // for code 4 type responses also the index is reported back
58 res = cosmos::sprintf("\033]4;%d;rgb:%02x%02x/%02x%02x/%02x%02x\007", cosmos::to_integral(idx), r, r, g, g, b, b);
59 } else {
60 res = cosmos::sprintf("\033]%d;rgb:%02x%02x/%02x%02x/%02x%02x\007", code, r, r, g, g, b, b);
61 }
62
63 if (res.empty()) {
64 // sprintf error occurred
65 m_nst.logger().error() << "sprintf failed while printing OSC (code " << code << ") response\n";
66 return;
67 }
68
69 m_nst.tty().write(res, TTY::MayEcho{true});
70}

88std::string StringEscape::joinArgs(const size_t start_idx) const {
89 std::string ret;
90
91 if (start_idx >= m_args.size())
92 return ret;
93
94 for (auto it = m_args.begin() + start_idx; it < m_args.end(); it++) {
95 auto &arg = *it;
96 if (arg.empty() || !arg[0])
97 return ret;
98
99 if (!ret.empty())
100 // add the original separator again
101 ret += ";";
102 ret += arg;
103 }
104
105 return ret;
106}

108void StringEscape::process() {
109 parseArgs();
110
111 switch (m_esc_type) {
112 case Type::OSC: // Operating System Command
113 if (!processOSC()) {
114 // error in OSC command, dump
115 dump("erresc: unknown str escape");
116 }
117 return;
118 case Type::SET_TITLE: // old title set compatibility
119 setTitle(m_args.empty() ? "" : joinArgs(0).c_str());
120 return;
121 case Type::DCS: //
122 case Type::APC: // Application Program Command
123 case Type::PM: // Privacy Message
124 case Type::NONE: // should never happen
125 return;
126 }

129bool StringEscape::processOSC() {
130 auto &wsys = m_nst.wsys();
131 const int par = m_args.empty() ? 0 : std::atoi(m_args[0].data());
132 const auto numargs = m_args.size();
133
134 // handles different color settings and reporting
135 auto handle_color = [&](const std::string_view label, const int code, const ColorIndex idx) {
136 if (numargs < 2)
137 return false;
138
139 const auto arg = m_args[1];
140
141 if (arg == "?")
142 // report current color setting
143 oscColorResponse(idx, code);
144 else if (!wsys.colorManager().setColorName(idx, arg.data()))
145 m_nst.logger().error() << "erresc: invalid " << label << " color: " << arg << "\n";
146 else
147 m_nst.term().setAllDirty();
148
149 return true;
150 };
151
152 constexpr int SET_COLOR_INDEX = 4;
153 constexpr int RESET_COLOR_INDEX = 104;
154
155 auto &theme = m_nst.theme();
156
157 // for reference see: https://www.xfree86.org/current/ctlseqs.html
158 switch (par) {
159 case 0: // change icon name _and_ window title
160 if (numargs > 1) {
161 const auto title = joinArgs(1);
162 setTitle(title.c_str());
163 setIconTitle(title.c_str());
164 }
165 break;
166 case 1: // change icon name
167 if (numargs > 1)
168 setIconTitle(joinArgs(1).c_str());
169 break;
170 case 2: // change window title
171 if (numargs > 1)
172 setTitle(joinArgs(1).c_str());
173 break;
174 case 52: // manipulate selection data
175 if (numargs > 2 && config::ALLOW_WINDOW_OPS) {
176 auto decoded = base64::decode(m_args[2]);
177 if (!decoded.empty()) {
178 wsys.selection().setSelection(decoded);
179 wsys.copyToClipboard();
180 } else {
181 m_nst.logger().error() << "erresc: invalid base64\n";
182 }
183 }
184 break;
185 case 10: // change text FG color
186 return handle_color("foreground", par, theme.fg);
187 case 11: // change text BG color
188 return handle_color("background", par, theme.bg);
189 case 12: // change text cursor color
190 return handle_color("cursor", par, theme.cursor_color);
191 case SET_COLOR_INDEX: // change color number to RGB value
192 if (numargs < 3)
193 return false;
194 /* FALLTHROUGH */
195 case RESET_COLOR_INDEX: { // color reset
196 const auto name = (par == SET_COLOR_INDEX ? m_args[2] : std::string_view(""));
197 const int rawindex = (numargs > 1) ? atoi(m_args[1].data()) : -1;
198 const auto colindex = rawindex >= 0 ? ColorIndex(rawindex) : ColorIndex::INVALID;
199
200 if (name == "?")
201 oscColorResponse(colindex, SET_COLOR_INDEX);
202 else if (!wsys.colorManager().setColorName(colindex, name.data())) {
203 if (par == RESET_COLOR_INDEX && numargs <= 1) {
204 wsys.resetColors();
205 break; // color reset without parameter
206 }
207 m_nst.logger().error() << "erresc: invalid color index=" << rawindex << ", name=" << (name.empty() ? "(null)" : name) << "\n";
208 } else {
209 m_nst.term().setAllDirty();
210 wsys.clearWindow();
211 }
212 break;
213 }
214 default: return false;
215 }
216
217 return true;
218}

221void StringEscape::parseArgs() {
222 auto it = m_str.begin();
223
224 // parameters are separated by semicloon, extract them
225
226 while (it != m_str.end() && m_args.size() < MAX_STR_ARGS) {
227 auto end = std::find(it, m_str.end(), ';');
228
229 // NOTE: C++20 has a better constructor using iterators
230 std::string_view sv(&(*it), std::distance(it, end));
231 m_args.push_back(sv);
232
233 if (end != m_str.end()) {
234 // make sure the views we add to m_args are properly terminated
235 *end = '\0';
236 end++;
237 }
238
239 it = end;
240 }
241
242 if (it != m_str.end()) {
243 m_nst.logger().error() << __FUNCTION__ << ": maximum number of arguments exceeded\n";
244 }
245}

247void StringEscape::dump(const std::string_view prefix) const {
248 auto &errlog = m_nst.logger().error();
249 errlog << prefix << " ESC" << static_cast<char>(m_esc_type);
250
251 for (const auto c: m_str) {
252 if (c == '\0') {
253 errlog << '\n';
254 return;
255 } else if (cosmos::printable(c)) {
256 errlog << (char)c;
257 } else if (c == '\n') {
258 errlog << "(\\n)";
259 } else if (c == '\r') {
260 errlog << "(\\r)";
261 } else if (c == 0x1b) {
262 errlog << "(\\e)";
263 } else {
264 errlog << "(" << cosmos::HexNum{static_cast<unsigned>(c), 2}.showBase(false) << ")";
265 }
266 }
267 errlog << "ESC\\\n";
268}

277void StringEscape::add(const std::string_view s) {
278 if (m_str.size() + s.size() >= m_str.capacity()) {
279 /*
280 * Here is a bug in terminals. If the user never sends
281 * some code to stop the str or esc command, then nst
282 * will stop responding. But this is better than
283 * silently failing with unknown characters. At least
284 * then users will report back.
285 *
286 * In the case users ever get fixed, here is the code:
287 */
288 /*
289 * m_nst.term().m_esc_state.reset();
290 * process();
291 */
292 if (m_str.size() > (SIZE_MAX - utf8::UTF_SIZE) / 2)
293 return;
294 m_str.reserve(m_str.capacity() << 1);
295 }
296
297 m_str.append(s);
298}
