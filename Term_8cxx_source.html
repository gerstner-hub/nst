121void Term::resize(const TermSize new_size) {
122 // remember the old size for updating new screen regions below
123 const auto old_size = m_size;
124
125 assert(new_size.valid());
126 if (!new_size.valid()) {
127 return;
128 }
129
130 // shift the screen view down to keep cursor where we expect it
131 //
132 // only do this if the current cursor row will be outside the limits
133 // of the new terminal size.
134 //
135 // moveCursorTo() below will update the cursor position accordingly to
136 // make things fit again.
137 if (const auto shift = m_cursor.pos.y - new_size.rows + 1; shift > 0) {
138 for (auto screen: {&m_screen, &m_saved_screen}) {
139 screen->shiftViewDown(shift);
140 }
141
142 m_selection.scroll(0, -shift);
143 }
144
145 m_screen.saveScrollState();
146
147 // adjust dimensions of internal data structures
148 m_screen.setDimension(new_size, m_cursor.attrs());
149 m_saved_screen.setDimension(new_size, m_saved_screen.getCachedCursor().attrs());
150
151 // update terminal size (needed by setupTabs() below)
152 m_size = new_size;
153
154 // recalculate tab markers if we have more columns now
155 if (new_size.cols > old_size.cols) {
156 setupTabs();
157 }
158
159 // reset scrolling region
160 resetScrollArea();
161 // make use of the clamping in moveCursorTo() to get a valid cursor position again
162 moveCursorTo(m_cursor.pos);
163
164 // Clear both screens (it marks all lines dirty)
165 const auto saved_cursor = m_cursor;
166
167 for (size_t i = 0; i < 2; i++) {
168 // clear new cols if number of columns increased
169 if (old_size.cols < new_size.cols && old_size.rows > 0) {
170 // we don't need to clear the Glyphs, they are
171 // initialized in Screen::setDimension()
172 setDirty(LineSpan{0, old_size.rows - 1});
173 }
174
175 // clear new rows if number of rows increased
176 if (old_size.rows < new_size.rows && old_size.cols > 0) {
177 clearRegion({CharPos{0, old_size.rows}, bottomRight()});
178 }
179 swapScreen();
180 cursorControl(CursorState::Control::LOAD);
181 }
182
183 m_cursor = saved_cursor;
184
185 // keep the current scrolling position even when changing the window
186 // size. note that this only works if terminal output is disabled
187 // (ctrl + s), otherwise e.g. the shell will attempt to reconstruct
188 // its prompt, triggering a stopScrolling() in write().
189 // to further improve this we would need to differentiate between user
190 // input and application output in write(), while still allowing
191 // application output to be added to the current screen without
192 // changing the scrolling position.
193 if (!m_screen.restoreScrollState()) {
194 scrollHistoryUpMax();
195 m_selection.reset();
196 }
197
198 if (m_screen.isScrolled()) {
199 setAllDirty();
200 }
201
202}

204void Term::clearRegion(Range range) {
205
206 range.sanitize();
207 range.clamp(bottomRight());
208
209 for (auto pos = range.begin; pos.y <= range.end.y; pos.y++) {
210 auto &line = m_screen[pos.y];
211 line.setDirty(true);
212 line.shrinkToPhysical();
213
214 if (line.empty()) {
215 // if this is a new line that has been scrolled into
216 // view then we need to set it to proper size first
217 line.resize(m_size.cols);
218 }
219
220 for (pos.x = range.begin.x; pos.x <= range.end.x; pos.x++) {
221 if (m_selection.isSelected(pos))
222 m_selection.reset();
223 auto &gp = m_screen[pos];
224 gp.clear(m_cursor.attrs());
225 }
226 }
227}

299void Term::setAltScreen(const bool enable, const bool with_cursor, const bool clear_alt) {
300 if (!m_allow_altscreen)
301 return;
302
303 // if the mode actually changed
304 const bool need_switch = enable ^ onAltScreen();
305
306 /*
307 * indeed original XTerm save/restores the cursor position again even
308 * if the screen is already set to the correct target
309 */
310
311 if (with_cursor && enable)
312 // save cursor before entering alt screen
313 cursorControl(CursorState::Control::SAVE);
314
315 if (need_switch) {
316 // one control clears only upon leaving the alt screen,
317 // another only upon entering it
318 if (clear_alt && onAltScreen())
319 clearScreen();
320 swapScreen();
321 if (clear_alt && onAltScreen())
322 clearScreen();
323 }
324
325 if (with_cursor && !enable)
326 // restore cursor when returning from alt screen
327 cursorControl(CursorState::Control::LOAD);
328}

388void Term::deleteColsAfterCursor(int count) {
389 count = std::clamp(count, 0, lineSpaceLeft());
390
391 const auto cursor = m_cursor.pos;
392 const int dst = cursor.x;
393 const int src = cursor.x + count;
394 const int left = m_size.cols - src;
395 auto &line = m_screen[cursor.y];
396
397 if (left <= 0)
398 return;
399
400 // NOTE: in C++20 there will be std::shift_left for this
401
402 // slide remaining line content count characters to the left
403 std::memmove(&line[dst], &line[src], left * sizeof(Line::value_type));
404 // clear count characters at end of line
405 clearRegion(Range{
406 CharPos{m_size.cols - count, cursor.y},
407 Width{count}
408 });
409
410 if (line[dst].isDummy()) {
411 // we delete only part of a wide character, shift the line for another column then
412 deleteColsAfterCursor(1);
413 }
414}

468void Term::scrollDown(int num_lines, std::optional<int> opt_origin) {
469 const auto area = m_scroll_area;
470 const int origin = opt_origin ? *opt_origin : area.top;
471
472 num_lines = std::clamp(num_lines, 0, area.bottom - origin + 1);
473
474 /* see scrollUp() */
475
476 for (auto i = area.bottom + 1; i < m_size.rows; i++) {
477 std::swap(m_screen[i], m_screen[i-num_lines]);
478 }
479
480 for (auto i = 0; i < origin; i++) {
481 std::swap(m_screen[i], m_screen[i-num_lines]);
482 }
483
484 m_screen.shiftViewUp(num_lines);
485
486 // clear the to-be-overwritten lines, which will end up at the top
487 // after scrolling finished below.
488 clearLines(LineSpan{origin, origin + num_lines - 1});
489 setDirty(LineSpan{origin, area.bottom});
490 m_selection.scroll(origin, num_lines);
491}
492
493void Term::scrollUp(int num_lines, std::optional<int> opt_origin) {
494 const auto area = m_scroll_area;
495 int origin = opt_origin ? *opt_origin : area.top;
496
497 num_lines = std::clamp(num_lines, 0, area.bottom - origin + 1);
498
499 /*
500 * shift non-scrolling top and bottom areas downward by `num_lines`.
501 * By doing so the upper `num_lines` from the scrolling area will
502 * become scroll history lines. The new bottom lines that are
503 * appearing will be the oldest lines found in the scroll history
504 * ring buffer, which will be cleared at the end.
505 *
506 * Finally the ring buffer view will be shifted by `num_lines`
507 * downwards to keep everything in place.
508 */
509
510 for (auto i = origin - 1; i >= 0; i--) {
511 std::swap(m_screen[i], m_screen[i+num_lines]);
512 }
513
514 for (auto i = m_size.rows - 1; i > area.bottom; i--) {
515 std::swap(m_screen[i], m_screen[i+num_lines]);
516 }
517
518 m_screen.shiftViewDown(num_lines);
519
520 clearLines(LineSpan{area.bottom - num_lines + 1, area.bottom});
521 setDirty(LineSpan{origin, area.bottom});
522 m_selection.scroll(origin, -num_lines);
523}

601void Term::drawScreen() const {
602
603 const Range range{topLeft(), bottomRight()};
604
605 for (int y = range.begin.y; y <= range.end.y; y++) {
606 auto &line = m_screen[y];
607
608 if (!line.isDirty())
609 continue;
610
611 // beware that we need to point past the last valid position,
612 // `range` has inclusive end coordinates!
613 m_wsys.drawGlyphs(
614 line.begin() + range.begin.x,
615 line.begin() + raw_width(Rect{range}.width()),
616 CharPos{range.begin.x, y});
617
618 line.setDirty(false);
619 }

622void Term::drawCursor() const {
623 /*
624 * we want to be able to draw the cursor even if we are currently
625 * scrolled back - if the cursor is still visible on the screen, that
626 * is.
627 *
628 * to achieve this we shift the cursor positions using the Screen
629 * helpers accordingly.
630 */
631 const auto orig_last_pos = m_screen.shiftedPos(m_last_cursor_pos);
632 auto shifted_last_pos = orig_last_pos;
633
634 CharPos new_pos;
635 if (auto shifted_new_pos = m_screen.shiftedPos(m_cursor.pos); shifted_new_pos) {
636 new_pos = *shifted_new_pos;
637 } else {
638 // cursor isn't visible
639 return;
640 }
641
642 // in case we point to a wide character dummy position, move one
643 // character to the left to point to the actual character
644 if (shifted_last_pos) {
645 clampToScreen(*shifted_last_pos);
646 if (m_screen[*shifted_last_pos].isDummy())
647 shifted_last_pos->moveLeft();
648 }
649
650 if (m_screen[new_pos].isDummy())
651 new_pos.moveLeft();
652
653 if (shifted_last_pos) {
654 m_wsys.clearCursor(*shifted_last_pos, m_screen[*shifted_last_pos]);
655 }
656 m_wsys.drawCursor(new_pos, m_screen[new_pos]);
657
658 if (auto new_unshifted_pos = m_screen.unshiftedPos(new_pos); new_unshifted_pos) {
659 m_last_cursor_pos = *new_unshifted_pos;
660 }
661
662 if (const bool cursor_pos_changed = orig_last_pos != new_pos; cursor_pos_changed) {
663 m_wsys.setInputSpot(new_pos);
664 }
665}

676Rune Term::translateChar(Rune rune) const {
677 // GRAPHIC0 translation table for VT100 "special graphics mode"
678 //
679 // The table is proudly stolen from rxvt.
680 //
681
682 constexpr auto VT100_GR_START = 0x41;
683 constexpr auto VT100_GR_END = 0x7e;
684
685 // these have to be strings, not characters, because these are
686 // multi-byte characters that don't fit into char.
687 constexpr std::string_view VT100_0[VT100_GR_END - VT100_GR_START + 1] = { // 0x41 - 0x7e
688 u8"↑", u8"↓", u8"→", u8"←", u8"█", u8"▚", u8"☃", // A - G
689 {}, {}, {}, {}, {}, {}, {}, {}, // H - O
690 {}, {}, {}, {}, {}, {}, {}, {}, // P - W
691 {}, {}, {}, {}, {}, {}, {}, u8" ", // X - _
692 u8"◆", u8"▒", u8"␉", u8"␌", u8"␍", u8"␊", u8"°", u8"±", // ` - g
693 u8"␤", u8"␋", u8"┘", u8"┐", u8"┌", u8"└", u8"┼", u8"⎺", // h - o
694 u8"⎻", u8"─", u8"⎼", u8"⎽", u8"├", u8"┤", u8"┴", u8"┬", // p - w
695 u8"│", u8"≤", u8"≥", u8"π", u8"≠", u8"£", u8"·", // x - ~
696 };
697
698 switch (m_charsets[m_active_charset]) {
699 // nothing to do or not implemented
700 default: break;
701 case Charset::GRAPHIC0:
702 if (cosmos::in_range(rune, VT100_GR_START, VT100_GR_END)) {
703 const auto TRANS_CHAR = VT100_0[rune - VT100_GR_START];
704
705 if (!TRANS_CHAR.empty()) {
706 utf8::decode(TRANS_CHAR, rune);
707 }
708 }
709 break;
710 }
711
712 return rune;
713}

715void Term::setChar(const Rune rune, const CharPos pos) {
716 auto &glyph = m_screen[pos];
717
718 // if we replace a WIDE/DUMMY position then correct the sibling position
719 if (glyph.isWide()) {
720 if (!isAtEndOfLine(pos)) {
721 auto &next_glyph = m_screen[pos.nextCol()];
722 next_glyph.rune = ' ';
723 next_glyph.resetDummy();
724 }
725 } else if (glyph.isDummy()) {
726 auto &prev_glyph = m_screen[pos.prevCol()];
727 prev_glyph.rune = ' ';
728 prev_glyph.resetWide();
729 }
730
731 m_screen[pos.y].setDirty(true);
732 glyph = m_cursor.attrs();
733 glyph.rune = translateChar(rune);
734}

757void Term::putChar(const Rune rune) {
758 const auto rinfo = RuneInfo{rune, m_mode[Term::Mode::UTF8]};
759
760 if (isPrintMode()) {
761 m_tty.printToIoFile(rinfo.encoded());
762 }
763
764 if (m_esc_handler.process(rinfo) == EscapeHandler::WasProcessed{true})
765 // input was part of a special control sequence
766 return;
767
768 if (m_selection.isSelected(m_cursor.pos))
769 m_selection.reset();
770
771 Glyph *gp = curGlyph();
772
773 // perform automatic line wrap, if necessary
774 if (m_mode[Mode::WRAP] && m_cursor.needWrapNext()) {
775 gp->setWrapped();
776 moveToNewline();
777 gp = curGlyph();
778 }
779
780 const auto req_width = rinfo.width();
781
782 if (const auto left = lineSpaceLeft(); left < req_width) {
783 // if this is a multi-column wide character that doesn't fit
784 // into this line anymore then clear remaining existing
785 // characters
786 clearColsAfterCursor();
787 moveToNewline();
788 gp = curGlyph();
789 }
790
791 // shift any remaining Glyphs to the right
792 if (m_mode[Mode::INSERT]) {
793 if (const auto to_move = lineSpaceLeft() - req_width; to_move > 0) {
794 std::memmove(gp + req_width, gp, to_move * sizeof(Line::value_type));
795 gp->mode.reset();
796 }
797 }
798
799 setChar(rune, m_cursor.pos);
800 m_last_char = rune;
801 const auto left_chars = lineSpaceLeft();
802
803 if (rinfo.isWide()) {
804 gp->setWide();
805
806 // if there's only one left character space then we'd be
807 // operating in a miniscule terminal size and cannot
808 // properly display wide characters.
809
810 if (left_chars > 1) {
811 // mark the follow-up position as dummy
812 auto &next = gp[1];
813
814 // if we are overriding another wide character, clean
815 // up the dummy follow-up
816 if (next.isWide() && left_chars > 2) {
817 auto &after_next = gp[2];
818 after_next.rune = ' ';
819 after_next.resetDummy();
820 }
821
822 next.makeDummy();
823 }
824 }
825
826 if (left_chars > req_width) {
827 moveCursorTo(m_cursor.pos.nextCol(req_width));
828 } else {
829 m_cursor.setWrapNext(true);
830 }
831}

833size_t Term::write(const std::string_view data, const ShowCtrlChars show_ctrl) {
834 Rune rune;
835 size_t charsize = 0;
836 const bool use_utf8 = m_mode[Mode::UTF8];
837
838 // jump back to the current input screen upon entering new data
839 //
840 // if it is non-interactive input then we will return to the
841 // scrolling position once any changes to the current screen
842 // are complete.
843 //
844 // this allows to stay scrolled back while new output data is
845 // appended.
846 //
847 // we also have to consider the current selection here. It will be
848 // shifted if e.g. a newline is processed. If we return to the
849 // previous scroll position then we also have to restore the selection
850 // coordinates.
851 const auto saved_scroll = m_screen.saveScrollState();
852 if (m_keep_scroll_position && saved_scroll) {
853 m_selection.saveRange();
854 }
855 m_screen.stopScrolling();
856
857 for (size_t pos = 0; pos < data.size(); pos += charsize) {
858 if (use_utf8) {
859 // process a complete utf8 char
860 charsize = utf8::decode(data.substr(pos), rune);
861 if (charsize == 0)
862 return pos;
863 } else {
864 rune = data[pos] & 0xFF;
865 charsize = 1;
866 }
867
868 if (show_ctrl && RuneInfo::isControlChar(rune)) {
869 // add symbolic annotation for control chars
870 if (rune & 0x80) {
871 rune &= 0x7f;
872 putChar('^');
873 putChar('[');
874 } else if (!cosmos::in_list(static_cast<char>(rune), {'\n', '\r', '\t'})) {
875 rune ^= 0x40;
876 putChar('^');
877 }
878 }
879
880 putChar(rune);
881 }
882
883 if (m_keep_scroll_position) {
884 if (!m_screen.restoreScrollState()) {
885 scrollHistoryUpMax();
886 } else if (saved_scroll) {
887 m_selection.restoreRange();
888 }
889 }
890
891 return data.size();
892}
