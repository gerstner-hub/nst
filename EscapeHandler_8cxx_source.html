27void EscapeHandler::handleControlCode(const RuneInfo &rinfo) {
28
29 auto &term = m_nst.term();
30 const auto &cursor = term.cursor();
31 const auto code = rinfo.asChar();
32
33 switch (code) {
34 case '\t': // HT
35 term.moveToNextTab();
36 return;
37 case '\b': // BS
38 term.moveCursorTo(cursor.position().prevCol());
39 return;
40 case '\r': // CR
41 term.moveCursorTo(cursor.position().startOfLine());
42 return;
43 case '\f': // LF
44 case '\v': // VT
45 case '\n': { // LF
46 // go also to first col if CRLF mode is set
47 term.moveToNewline(term.carriageReturn());
48 return;
49 }
50 case '\a': { // BEL
51 // backwards compatibility to xterm, which also accepts BEL
52 // (instead of 'ST') as OSC command terminator.
53 const auto handled = handleCommandTerminator();
54 if (!handled)
55 // otherwise process as a regular bell
56 m_nst.wsys().ringBell();
57 break;
58 }
59 case '\033': // ESC
60 markNewCSI();
61 return;
62 case '\016': // SO (LS1 -- Locking shift 1)
63 case '\017': { // SI (LS0 -- Locking shift 0)
64 // switch between predefined character sets
65 const auto charset = 1 - (code - '\016');
66 term.setCharset(charset);
67 return;
68 }
69 case '\032': // SUB
70 term.showSubMarker();
71 /* FALLTHROUGH */
72 case '\030': // CAN
73 m_csi_escape.reset();
74 break;
75 case '\005': // ENQ (IGNORED)
76 case '\000': // NUL (IGNORED)
77 case '\021': // XON (IGNORED)
78 case '\023': // XOFF (IGNORED)
79 case 0177: // DEL (IGNORED)
80 return;
81 case 0x80: // TODO: PAD
82 case 0x81: // TODO: HOP
83 case 0x82: // TODO: BPH
84 case 0x83: // TODO: NBH
85 case 0x84: // TODO: IND
86 break;
87 case 0x85: // NEL -- Next line
88 // always go to first col
89 term.moveToNewline(Term::CarriageReturn{true});
90 break;
91 case 0x86: // TODO: SSA
92 case 0x87: // TODO: ESA
93 break;
94 case 0x88: // HTS -- Horizontal tab stop
95 term.setTabAtCursor(true);
96 break;
97 case 0x89: // TODO: HTJ
98 case 0x8a: // TODO: VTS
99 case 0x8b: // TODO: PLD
100 case 0x8c: // TODO: PLU
101 case 0x8d: // TODO: RI
102 case 0x8e: // TODO: SS2
103 case 0x8f: // TODO: SS3
104 case 0x91: // TODO: PU1
105 case 0x92: // TODO: PU2
106 case 0x93: // TODO: STS
107 case 0x94: // TODO: CCH
108 case 0x95: // TODO: MW
109 case 0x96: // TODO: SPA
110 case 0x97: // TODO: EPA
111 case 0x98: // TODO: SOS
112 case 0x99: // TODO: SGCI
113 break;
114 case 0x9a: // DECID -- Identify Terminal
115 m_nst.tty().write(config::VT_IDENT, TTY::MayEcho{false});
116 break;
117 case 0x9b: // TODO: CSI
118 case 0x9c: // TODO: ST
119 break;
120 case 0x90: // DCS -- Device Control String
121 case 0x9d: // OSC -- Operating System Command
122 case 0x9e: // PM -- Privacy Message
123 case 0x9f: { // APC -- Application Program Command
124 const auto esc_type = static_cast<StringEscape::Type>(code);
125 initStringEscape(esc_type);
126 return;
127 }
128 } // end switch
129
130 // only CAN, SUB, \a and C1 chars interrupt a sequence
131 resetStringEscape();
132}

134EscapeHandler::WasProcessed EscapeHandler::process(const RuneInfo &rinfo) {
135
136 // STR sequence must be checked before anything else because it uses
137 // all following characters until it receives an ESC, SUB, ST or
138 // any other C1 control character.
139 if (inStringEscape()) {
140 if (m_str_escape.isTerminator(rinfo)) {
141 // NOTE: this is a bit of weird spot here, we're not
142 // returning yet, but process the actual terminator
143 // further down below. Since ST consists of two bytes
144 // 'ESC \', the actual StringEscape completion can be
145 // parsed in handleInitialEscape(). Alternatively a BEL
146 // character is also supported which is parsed in
147 // handleControlCode()
148 markStringEscapeFinal();
149 } else {
150 m_str_escape.add(rinfo.encoded());
151 return WasProcessed{true};
152 }
153 }
154
155 // Actions of control codes must be performed as soon they arrive
156 // because they can be embedded inside a control sequence, and they
157 // must not cause conflicts with sequences.
158 if (rinfo.isControlChar()) {
159 auto &term = m_nst.term();
160
161 // in UTF-8 mode ignore handling C1 control characters
162 if (term.mode()[Term::Mode::UTF8] && rinfo.isControlC1())
163 return WasProcessed{true};
164 handleControlCode(rinfo);
165 // control codes are not shown ever
166 if (m_state.none())
167 term.resetLastChar();
168
169 return WasProcessed{true};
170 } else if (m_state[Escape::START]) {
171 const bool finished = checkCSISequence(rinfo);
172
173 if (finished) {
174 m_state.reset();
175 }
176
177 // All characters which form part of a sequence are not printed
178 return WasProcessed{true};
179 }
180
181 return WasProcessed{false};
182}

184bool EscapeHandler::checkCSISequence(const RuneInfo &rinfo) {
185 const auto rune = rinfo.rune();
186 auto &term = m_nst.term();
187
188 if (m_state[Escape::CSI]) {
189 const bool finished = m_csi_escape.addCSI(rinfo.rune());
190 if (finished) {
191 // reset the state right away to prevent a potential
192 // infinite recursion (e.g. via the CSI repeat char
193 // logic).
194 m_state.reset();
195 m_csi_escape.parse();
196 m_csi_escape.process();
197 }
198 return finished;
199 } else if (m_state[Escape::UTF8]) {
200 switch (rune) {
201 case 'G':
202 term.setUTF8(true);
203 break;
204 case '@':
205 term.setUTF8(false);
206 break;
207 }
208 } else if (m_state[Escape::ALTCHARSET]) {
209 // this is DEC VT100 spec related
210 switch(rune) {
211 default:
212 m_nst.logger().error() << "esc unhandled charset: ESC ( " << rune << ")\n";
213 break;
214 case '0':
215 term.setCharsetMapping(m_esc_charset, Term::Charset::GRAPHIC0);
216 break;
217 case 'B':
218 term.setCharsetMapping(m_esc_charset, Term::Charset::USA);
219 break;
220 }
221 } else if (m_state[Escape::TEST]) {
222 if (rune == '8') {
223 term.runDECTest();
224 }
225 } else {
226 auto state = handleInitialEscape(rune);
227
228 if (state) {
229 m_state.set(*state);
230 // sequence not yet finished
231 return false;
232 }
233 }
234
235 return true;
236}

238std::optional<EscapeHandler::Escape> EscapeHandler::handleInitialEscape(const char ch) {
239 auto &term = m_nst.term();
240 auto &wsys = m_nst.wsys();
241
242 // for reference see `man 4 console_codes`
243
244 // these are, apart from '[', non-CSI escape sequences that we handle
245 // directly in this class, CSI is handled by m_csi_escape
246 switch (ch) {
247 case '[':
248 return Escape::CSI;
249 case '#':
250 return Escape::TEST;
251 case '%': // character set selection
252 return Escape::UTF8;
253 case 'P': // DCS -- Device Control String
254 case '_': // APC -- Application Program Command
255 case '^': // PM -- Privacy Message
256 case ']': // OSC -- Operating System Command
257 case 'k': { // old title set compatibility
258 // hand over to StringEscape
259 const auto esc_type = StringEscape::Type{ch};
260 initStringEscape(esc_type);
261 return Escape::STR;
262 }
263 case 'n': // LS2 -- Locking shift 2
264 case 'o': // LS3 -- Locking shift 3
265 term.setCharset(2 + (ch - 'n'));
266 break;
267 case '(': // GZD4 -- set primary charset G0
268 case ')': // G1D4 -- set secondary charset G1
269 case '*': // G2D4 -- set tertiary charset G2
270 case '+': // G3D4 -- set quaternary charset G3
271 m_esc_charset = ch - '(';
272 return Escape::ALTCHARSET;
273 case 'D': // IND -- Linefeed
274 term.doLineFeed();
275 break;
276 case 'E': // NEL -- Next line
277 term.moveToNewline(Term::CarriageReturn{true}); // always go to first col
278 break;
279 case 'H': // HTS -- Horizontal tab stop
280 term.setTabAtCursor(true);
281 break;
282 case 'M': // RI -- Reverse index / linefeed
283 term.doReverseLineFeed();
284 break;
285 case 'Z': // DECID -- Identify Terminal
286 m_nst.tty().write(config::VT_IDENT, TTY::MayEcho{false});
287 break;
288 case 'c': // RIS -- Reset to initial state
289 term.reset();
290 wsys.resetState();
291 break;
292 case '=': // DECPAM -- Application keypad
293 wsys.setMode(WinMode::APPKEYPAD, true);
294 break;
295 case '>': // DECPNM -- Normal keypad
296 wsys.setMode(WinMode::APPKEYPAD, false);
297 break;
298 case '7': // DECSC -- Save Cursor
299 term.cursorControl(CursorState::Control::SAVE);
300 break;
301 case '8': // DECRC -- Restore Cursor
302 term.cursorControl(CursorState::Control::LOAD);
303 break;
304 case '\\': // ST -- String Terminator for StringEscape (!)
305 /* this likely is the second byte of the ST := ESC \ */
306 handleCommandTerminator();
307 break;
308 default:
309 m_nst.logger().error() << "erresc: unknown sequence ESC " << cosmos::HexNum(ch, 2)
310 << " '" << (cosmos::printable(ch) ? ch : '.') << "'\n";
311 break;
312 }
313
314 return {};
315}
